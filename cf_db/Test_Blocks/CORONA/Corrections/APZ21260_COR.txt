! Patch to make it possible to run CORONA on APZ 212 60 and APZ 212 70;

PCORL:BLOCK=CORONA,IA=H'00BF,SUID="7000/CAA 107 9999/BROMIDE R5A02",CI=APZ21260;
  LHC  CR/W0-60;
  JER  AR0,H'010B;
  LHC  CR/W0-70;
  JER  AR0,H'010B;
END;
PCORI:BLOCK=CORONA,IA=H'00BF;

@T 5

PCORS:BLOCK=CORONA,IA=H'00BF;

! To avoid having to do a restart, set the following variables;
TEST SYSTEM;
SET VAR CORONA 50=2;
SET VAR CORONA 17=1;
SET VAR CORONA 79=0;
SET VAR CORONA 94=1;
SET VAR CORONA 24=0;
END TEST;
